BEGIN;
CREATE TABLE "UsersInformation" (

    "type_doc" varchar(6) NOT NULL,

    "phone" varchar(20) NOT NULL,

    "matricula" varchar(30) NOT NULL,
    "paciente_id" integer NOT NULL UNIQUE REFERENCES "auth_user" ("id") DEFERRABLE INITIALLY DEFERRED
)
